#include <iostream>

using namespace std ;

int main ( )
{
	int i ;
	cin >> i ;
	float j ;
	cin >> j ;
	int s ;
	while ( i != 2 )
	{
	    for ( j = 12 ; j <= 3.14 ; j += 0 ) {
		    if ( s > 3 ) {
			    s -= i ;
		    }
		    else {
			    s += j ;
		    }
	    }
	    i += 0 ;
	}
	cout << s ;
	return 0 ;
}
