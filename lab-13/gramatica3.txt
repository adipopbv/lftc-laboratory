<program> ::= { <lista_instructiuni> return 0 ; }
<lista_instructiuni> ::= <instructiune> <lista_instructiuni>
<lista_instructiuni> ::= <instructiune>
<instructiune> ::= <decl> ;
<instructiune> ::= <if>
<decl> ::= <tip_de_date> <variabile>
<tip_de_date> ::= int
<if> ::= if ( <cond> ) { <lista_instructiuni> }
<cond> ::= <variabile> <op_rel> <CONST>
<variabile> ::= <ID>
<ID> ::= a
<op_rel> ::= ==
<CONST> ::= 0
